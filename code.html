function getValues() {
  // get values from the page
  let startValue = document.getElementById("startValue").value;
  let endValue = document.getElementById("endValue").value;


  // Input Validation
  // parse into Integers
  startValue = parseInt(startValue);
  endValue = parseInt(endValue);

  if (Number.isInteger(startValue) && Number.isInteger(endValue)) {

    // We call generateNumbers
    let numbers = generateNumbers(startValue, endValue);

    // We call displayNumbers
    displayNumbers(numbers);
    
  } else {
    alert("You must enter integer/whole number");
  }
}
getValues( ) function:

This function is divided into smaller sub-functions

The function collects inputs from the DOM and parses them using the built-in parseInt method. If the string is successfully parsed, the function returns the integer value. If not, it returns NaN. If both values are integers, the generateNumbers(sValue, eValue) function is called. Otherwise, it shows an alert to the user to enter integer inputs

            
function generateNumbers(sValue, eValue) {
  let numbers = [];

  for (let index = sValue; index <= eValue; index++) {
    numbers.push(index);
  }
  return numbers;
}
generateNumbers(sValue, eValue)

This function generates an array of numbers between the start and end values using a loop. Finally, the displayNumbers(numbers) method is called

          
let templateRows = "";

function displayNumbers(numbers) {
  for (let index = 0; index < numbers.length; index++) {
    let className = "";
    let number = numbers[index];

    if (number % 2 === 0) {
      className = "even";
    } else {
      className = "odd";
    }

    // This DOESN'T work well with Prism. See the source
    templateRows += `${number}`;
  }

  document.getElementById("results").innerHTML = templateRows;
}
displayNumbers(numbers) function:

Here using a loop, we iterate through the numbers and check whether the number is odd or even.
